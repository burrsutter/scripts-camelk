apiVersion: camel.apache.org/v1alpha1
kind: Integration
metadata:
  creationTimestamp: null
  name: feed
  namespace: mycamel
spec:
  sources:
  - content: |-
      from('timer:tick?period=1s')
        .setBody().constant('Burr Wuz Here')
        .to('log:info')
        .to('knative:channel/messages')
    name: feed.groovy
status: {}


kamel run hello.groovy
